import java.util.Arrays;
import java.util.Objects;

/**
 * An array-based implementation of a linear list with dynamic resizing.
 * Includes methods for resizing, setting elements, clearing, removing ranges,
 * finding the last index of an element, and a dynamically resizing add method.
 *
 * @param <E> The type of elements in the list.
 */
public class ArrayLinearList<E> {
    private static final int DEFAULT_CAPACITY = 10;
    protected Object[] elementData;
    protected int size;

    /**
     * Constructs a list with the specified initial capacity.
     * @param initialCapacity The initial capacity of the list.
     * @throws IllegalArgumentException if the initial capacity is negative.
     */
    public ArrayLinearList(int initialCapacity) {
        if (initialCapacity < 0) {
            throw new IllegalArgumentException("Illegal Capacity: " + initialCapacity);
        }
        this.elementData = new Object[initialCapacity];
        this.size = 0;
    }

    /**
     * Constructs an empty list with a default capacity of 10.
     */
    public ArrayLinearList() {
        this(DEFAULT_CAPACITY);
    }

    /**
     * Checks if the given index is within the valid range of the list.
     * @param index The index to check.
     * @throws IndexOutOfBoundsException if the index is out of range.
     */
    private void checkIndex(int index) {
        if (index < 0 || index >= size) {
            throw new IndexOutOfBoundsException("Index: " + index + ", Size: " + size);
        }
    }

    /**
     * Retrieves the element at the specified index.
     * @param index The index of the element.
     * @return The element at the specified index.
     */
    public E get(int index) {
        checkIndex(index);
        return (E) elementData[index];
    }

    /**
     * Returns the number of elements in the list.
     * @return The size of the list.
     */
    public int size() {
        return size;
    }

    /**
     * Adds an element at the specified index, resizing the array if necessary.
     * @param index The index where the element is to be inserted.
     * @param theElement The element to add.
     * @throws IndexOutOfBoundsException if the index is out of range.
     */
    public void add(int index, E theElement) {
        if (index < 0 || index > size) {
            throw new IndexOutOfBoundsException("Index: " + index + ", Size: " + size);
        }

        // Resize array if it's full. Capacity is doubled.
        if (size == elementData.length) {
            int newCapacity = elementData.length == 0 ? 1 : elementData.length * 2;
            elementData = Arrays.copyOf(elementData, newCapacity);
        }

        // Shift elements to the right to make space for the new element.
        System.arraycopy(elementData, index, elementData, index + 1, size - index);

        elementData[index] = theElement;
        size++;
    }
    
    /**
     * Adds an element to the end of the list.
     * @param theElement The element to add.
     */
    public void add(E theElement) {
        add(size, theElement);
    }

    /**
     * Sets the size of the list.
     * If the new size is larger than the current size, null elements are added.
     * If the new size is smaller, elements are removed.
     * @param newSize The new size of the list.
     * @throws IllegalArgumentException if the new size is negative.
     */
    public void setSize(int newSize) {
        if (newSize < 0) {
            throw new IllegalArgumentException("New size cannot be negative.");
        }

        if (newSize > elementData.length) {
            elementData = Arrays.copyOf(elementData, newSize);
        } else if (newSize < size) {
            // Clear references of elements being removed for garbage collection.
            for (int i = newSize; i < size; i++) {
                elementData[i] = null;
            }
        }
        size = newSize;
    }

    /**
     * Replaces the element at the specified index with a new one.
     * @param theIndex The index of the element to replace.
     * @param theElement The new element.
     * @return The old element that was at the specified index.
     * @throws IndexOutOfBoundsException if the index is out of range.
     */
    public E set(int theIndex, E theElement) {
        checkIndex(theIndex);
        E oldElement = (E) elementData[theIndex];
        elementData[theIndex] = theElement;
        return oldElement;
    }

    /**
     * Removes all elements from the list, leaving it empty.
     */
    public void clear() {
        // Clear all elements for garbage collection.
        for (int i = 0; i < size; i++) {
            elementData[i] = null;
        }
        size = 0;
    }

    /**
     * Removes all elements in the specified range.
     * @param fromIndex The starting index of the range (inclusive).
     * @param toIndex The ending index of the range (exclusive).
     * @throws IndexOutOfBoundsException if the range is invalid.
     */
    public void removeRange(int fromIndex, int toIndex) {
        if (fromIndex < 0 || fromIndex > toIndex || toIndex > size) {
            throw new IndexOutOfBoundsException("Invalid range: [" + fromIndex + ", " + toIndex + ")");
        }

        int numRemoved = toIndex - fromIndex;
        if (numRemoved > 0) {
            // Shift elements to the left to fill the gap.
            System.arraycopy(elementData, toIndex, elementData, fromIndex, size - toIndex);

            // Clear references of the removed elements for garbage collection.
            int newSize = size - numRemoved;
            for (int i = newSize; i < size; i++) {
                elementData[i] = null;
            }
            size = newSize;
        }
    }

    /**
     * Returns the index of the last (rightmost) occurrence of the specified object.
     * @param o The object to search for.
     * @return The index of the last occurrence, or -1 if the object is not found.
     */
    public int lastIndexOf(Object o) {
        // Search backwards from the end of the list.
        for (int i = size - 1; i >= 0; i--) {
            if (Objects.equals(o, elementData[i])) {
                return i;
            }
        }
        return -1;
    }

    @Override
    public String toString() {
        Object[] temp = Arrays.copyOf(elementData, size);
        return Arrays.toString(temp);
    }

    public static void main(String[] args) {
        ArrayLinearList<String> list = new ArrayLinearList<>();
        list.add(0, "A");
        list.add(1, "B");
        list.add(2, "C");
        list.add(3, "A");
        list.add(4, "D");

        System.out.println("Original list: " + list); // Output: [A, B, C, A, D]

        // 1. setSize
        list.setSize(3);
        System.out.println("After setSize(3): " + list); // Output: [A, B, C]
        
        list.setSize(6);
        System.out.println("After setSize(6): " + list); // Output: [A, B, C, null, null, null]

        // 2. set
        System.out.println("Old element at index 1: " + list.set(1, "Z")); // Output: Old element at index 1: B
        System.out.println("After set(1, 'Z'): " + list); // Output: [A, Z, C, null, null, null]
        try {
            list.set(10, "Invalid");
        } catch (IndexOutOfBoundsException e) {
            System.out.println("Exception caught as expected: " + e.getMessage());
        }

        // 3. clear
        list.clear();
        System.out.println("After clear(): " + list); // Output: []

        // Repopulate for next test
        list.add(0, "X");
        list.add(1, "Y");
        list.add(2, "Z");
        list.add(3, "W");
        list.add(4, "V");
        System.out.println("\nRepopulated list: " + list); // Output: [X, Y, Z, W, V]

        // 4. removeRange
        list.removeRange(1, 3);
        System.out.println("After removeRange(1, 3): " + list); // Output: [X, W, V]
        try {
            list.removeRange(1, 10);
        } catch (IndexOutOfBoundsException e) {
            System.out.println("Exception caught as expected: " + e.getMessage());
        }

        // 5. lastIndexOf
        System.out.println("\nList for lastIndexOf: " + list); // Output: [X, W, V]
        list.add(3, "X");
        System.out.println("Updated list for lastIndexOf: " + list); // Output: [X, W, V, X]
        System.out.println("Last index of 'X': " + list.lastIndexOf("X")); // Output: 3
        System.out.println("Last index of 'Q': " + list.lastIndexOf("Q")); // Output: -1
        System.out.println("Last index of null: " + list.lastIndexOf(null)); // Output: -1
    }
}
